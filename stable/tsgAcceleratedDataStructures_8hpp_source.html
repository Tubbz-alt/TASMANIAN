691  ChangeType favorSparse(bool favor){
692  AlgorithmPreference new_preference = [=]()->AlgorithmPreference{
693  if (favor){
694  return (algorithm_select == algorithm_dense) ? algorithm_autoselect : algorithm_sparse;

726  ChangeType enable(TypeAcceleration acc, int new_gpu_id){
727  // get the effective new acceleration mode (use the fallback if acc is not enabled)
728  TypeAcceleration effective_acc = AccelerationMeta::getAvailableFallback(acc);
729  // if switching to a GPU mode, check if the device id is valid
730  if (AccelerationMeta::isAccTypeGPU(effective_acc) and ((new_gpu_id < 0 or new_gpu_id >= AccelerationMeta::getNumGpuDevices())))
731  throw std::runtime_error("Invalid CUDA device ID, see ./tasgrid -v for list of detected devices.");
732 
733  // assign the new values for the mode and device, but remember the current gpu state and check whether something changed
734  bool was_on_gpu = on_gpu();
735  ChangeType mode_change = (effective_acc == Utils::exchange(mode, effective_acc)) ? change_none : change_cpu_blas;
736  ChangeType device_change = (new_gpu_id == Utils::exchange(device, new_gpu_id)) ? change_none : change_gpu_device;
737 
738  if (AccelerationMeta::isAccTypeGPU(mode)){
739  // if the new mode is GPU-based, reset the engine and the handles (if already created)
740  engine = Utils::make_unique<GpuEngine>();
741  }else{
742  engine.reset();
743  }
744 
745  if (was_on_gpu){
746  // switching from gpu to gpu issues a change if the device is different
747  // switching from gpu to non-gpu acts as if the device has changed
748  return (on_gpu()) ? device_change : change_gpu_device;
749  }else{
750  // switching from non-gpu to gpu triggers gpu-enabled switch
751  // switching from non-gpu to non-gpu triggers blas change if the modes were different
752  return (on_gpu()) ? change_gpu_enabled : mode_change;
753  }
754  }
756  void setDevice() const{ AccelerationMeta::setDefaultCudaDevice(device); }
758  operator GpuEngine* () const{ return engine.get(); }
760  bool on_gpu() const{ return !!engine; }
